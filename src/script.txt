GAME MiPrimerJuego

ENTITY jugador {
    posX: 0,
    posY: 0,
    vida: 0,
    puntos: 0
}

ENTITY enemigo {
    posX: 10,
    posY: 10,
    vida: 0
}

ENTITY item {
    posX: 5,
    posY: 5,
    valor: 10
}

RULE ON_START {
    INPUT jugador.vida "Ingresa vida inicial del jugador:";
    INPUT enemigo.vida "Ingresa vida del enemigo:";
    PRINT "Bienvenido al juego!";
    PRINT "Tu mision es recolectar items y evitar enemigos";
    SET jugador.posX = 0;
    SET jugador.posY = 0;
}

RULE ON_UPDATE {
    MOVE jugador 5 0;
    
    IF (jugador.posX >= item.posX) {
        SET jugador.puntos = jugador.puntos + item.valor;
        PRINT "Item recolectado!";
        SET item.posX = item.posX + 10;
    }
    
    MOVE jugador 10 0;
    
    IF (jugador.posX >= enemigo.posX) {
        SET jugador.vida = jugador.vida - 10;
        PRINT "Golpeado por enemigo!";
    }
}

RULE ON_WIN {
    PRINT "Felicidades! Has ganado!";
    END_GAME;
}

RULE ON_LOSE {
    PRINT "Game Over";
    PRINT "Mejor suerte la proxima vez";
    END_GAME;
}
